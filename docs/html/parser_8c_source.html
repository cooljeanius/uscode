02239                         }

02271                                 error("expected end of expression", tokens);

02274                                 deleteBlockNodeList(blocks);
02275                                 return NULL;
02276                         }

02294                         error("expected end of expression", tokens);

02303                         deleteBlockNode(yes);

02321 #ifdef DEBUG
02322                 debug("ST_SWITCH");

02399                                 deleteExprNodeList(guards);
02400                                 deleteBlockNodeList(blocks);
02401                         }

02407                         }

02410                         if (!acceptToken(&tokens, TT_NEWLINE)) {
02411                                 error("expected end of expression", tokens);
02412                                 deleteExprNodeList(guards);
02413                                 deleteBlockNodeList(blocks);
02414                                 return NULL;

02418                                 deleteExprNodeList(guards);
02419                                 deleteBlockNodeList(blocks);
02420                                 return NULL;
02421                         }
02422                 }
02423                 if (!acceptToken(&tokens, TT_OIC)) {
02424                         error("expected OIC", tokens);
02425                         deleteExprNodeList(guards);
02426                         deleteBlockNodeList(blocks);
02427                         if (def) deleteBlockNode(def);
02428                         return NULL;
02429                 }
02430                 if (!acceptToken(&tokens, TT_NEWLINE)) {
02431                         error("expected end of expression", tokens);
02432                         deleteExprNodeList(guards);
02433                         deleteBlockNodeList(blocks);
02434                         if (def) deleteBlockNode(def);
02435                         return NULL;
02436                 }
02437                 stmt = createSwitchStmtNode(guards, blocks, def);
02438                 if (!stmt) {
02439                         deleteExprNodeList(guards);
02440                         deleteBlockNodeList(blocks);
02441                         if (def) deleteBlockNode(def);
02442                         return NULL;
02443                 }
02444                 ret = createStmtNode(ST_SWITCH, stmt);
02445                 if (!ret) {
02446                         deleteSwitchStmtNode(stmt);
02447                 }

02449         /* Break */
02450         else if (acceptToken(&tokens, TT_GTFO)) {
02451 #ifdef DEBUG
02452                 debug("ST_BREAK");
02453 #endif
02454                 if (!acceptToken(&tokens, TT_NEWLINE)) {
02455                         error("expected end of expression", tokens);
02456                         return NULL;
02457                 }
02458                 ret = createStmtNode(ST_BREAK, NULL);
02459                 if (!ret) return NULL;
02460         }
02461         /* Return */
02462         else if (acceptToken(&tokens, TT_FOUNDYR)) {
02463                 ExprNode *value = NULL;
02464                 ReturnStmtNode *stmt = NULL;
02465 #ifdef DEBUG
02466                 debug("ST_RETURN");
02467 #endif
02468                 value = parseExprNode(&tokens, functab);
02469                 if (!value) return NULL;
02470                 if (!acceptToken(&tokens, TT_NEWLINE)) {
02471                         error("expected end of expression", tokens);
02472                         deleteExprNode(value);
02473                         return NULL;
02474                 }
02475                 stmt = createReturnStmtNode(value);
02476                 if (!stmt) {
02477                         deleteExprNode(value);
02478                         return NULL;
02479                 }
02480                 ret = createStmtNode(ST_RETURN, stmt);
02481                 if (!ret) {
02482                         deleteReturnStmtNode(stmt);
02483                         return NULL;
02484                 }
02485         }
02486         /* Loop */
02487         else if (acceptToken(&tokens, TT_IMINYR)) {
02488                 IdentifierNode *name1 = NULL;
02489                 IdentifierNode *var = NULL;
02490                 ExprNode *update = NULL;
02491                 ExprNode *guard = NULL;
02492                 BlockNode *body = NULL;
02493                 FuncDefStmtNode *def = NULL;
02494                 IdentifierNode *name2 = NULL;
02495                 LoopStmtNode *stmt = NULL;
02496 #ifdef DEBUG
02497                 debug("ST_LOOP");
02498 #endif
02499                 name1 = parseIdentifierNode(&tokens);
02500                 if (!name1) return NULL;
02501                 if (peekToken(&tokens, TT_UPPIN) || peekToken(&tokens, TT_NERFIN)) {
02502                         OpType type;
02503                         IdentifierNode *varcopy = NULL;
02504                         ExprNode *arg1 = NULL;
02505                         ConstantNode *one = NULL;
02506                         ExprNode *arg2 = NULL;
02507                         ExprNodeList *args = NULL;
02508                         OpExprNode *op = NULL;
02509                         if (acceptToken(&tokens, TT_UPPIN)) {
02510                                 type = OP_ADD;
02511 #ifdef DEBUG
02512                         shiftout();
02513                         debug("ET_OP (OP_ADD)");
02514 #endif
02515                         }
02516                         else if (acceptToken(&tokens, TT_NERFIN)) {
02517                                 type = OP_SUB;
02518 #ifdef DEBUG
02519                         shiftout();
02520                         debug("ET_OP (OP_SUB)");
02521 #endif
02522                         }
02523                         else {
02524                                 error("expected UPPIN or NERFIN", tokens);
02525                                 return NULL;
02526                         }
02527                         if (!acceptToken(&tokens, TT_YR)) {
02528                                 error("expected YR", tokens);
02529                                 deleteIdentifierNode(name1);
02530                                 return NULL;
02531                         }
02532                         var = parseIdentifierNode(&tokens);
02533                         if (!var) {
02534                                 deleteIdentifierNode(name1);
02535                                 return NULL;
02536                         }
02537 #ifdef DEBUG
02538                         shiftout();
02539                         debug("ET_CONSTANT (CT_INTEGER)");
02540                         shiftin();
02541                         shiftin();
02542 #endif

02544                         if (!varcopy) {
02545                                 deleteIdentifierNode(name1);
02546                                 deleteIdentifierNode(var);
02547                                 return NULL;
02548                         }

02550                         if (!arg1) {
02551                                 deleteIdentifierNode(name1);
02552                                 deleteIdentifierNode(var);
02553                                 deleteIdentifierNode(varcopy);
02554                                 return NULL;
02555                         }
02556                         one = createIntegerConstantNode(1);
02557                         if (!one) {
02558                                 deleteIdentifierNode(name1);
02559                                 deleteIdentifierNode(var);
02560                                 deleteExprNode(arg1);
02561                                 return NULL;
02562                         }
02563                         arg2 = createExprNode(ET_CONSTANT, one);
02564                         if (!arg2) {
02565                                 deleteIdentifierNode(name1);
02566                                 deleteIdentifierNode(var);
02567                                 deleteExprNode(arg1);
02568                                 deleteConstantNode(one);
02569                                 return NULL;
02570                         }
02571                         args = createExprNodeList();
02572                         if (!args) {
02573                                 deleteIdentifierNode(name1);
02574                                 deleteIdentifierNode(var);
02575                                 deleteExprNode(arg1);
02576                                 deleteExprNode(arg2);
02577                                 return NULL;
02578                         }
02579                         if (!addExprNode(args, arg1)) {
02580                                 deleteIdentifierNode(name1);
02581                                 deleteIdentifierNode(var);
02582                                 deleteExprNode(arg1);
02583                                 deleteExprNode(arg2);
02584                                 deleteExprNodeList(args);
02585                                 return NULL;
02586                         }

02588                                 deleteIdentifierNode(name1);
02589                                 deleteIdentifierNode(var);
02590                                 deleteExprNode(arg2);
02591                                 deleteExprNodeList(args);
02592                                 return NULL;
02593                         }

02595                         if (!op) {
02596                                 deleteIdentifierNode(name1);
02597                                 deleteIdentifierNode(var);
02598                                 deleteExprNodeList(args);
02599                                 return NULL;
02600                         }
02601                         update = createExprNode(ET_OP, op);
02602                         if (!update) {
02603                                 deleteIdentifierNode(name1);
02604                                 deleteIdentifierNode(var);
02605                                 deleteOpExprNode(op);
02606                                 return NULL;
02607                         }
02608                 }
02609                 else if ((def = lookupFuncDefStmtNode(functab, (*tokens)->image))) {
02610                         tokens++;
02611                         if (!acceptToken(&tokens, TT_YR)) {
02612                                 error("expected YR", tokens);
02613                                 deleteIdentifierNode(name1);
02614                                 return NULL;
02615                         }
02616                         /* Check for unary function */
02617                         if (def->args->num == 1) {
02618                                 IdentifierNode *varcopy = NULL;
02619                                 ExprNode *arg = NULL;
02620                                 ExprNodeList *args = NULL;
02621                                 FuncCallExprNode *func = NULL;
02622 #ifdef DEBUG

02625 #endif
02626                                 var = parseIdentifierNode(&tokens);

02628                                         deleteIdentifierNode(name1);
02629                                         return NULL;
02630                                 }
02631 #ifdef DEBUG

02633 #endif

02635                                 if (!varcopy) {
02636                                         deleteIdentifierNode(name1);
02637                                         deleteIdentifierNode(var);
02638                                         return NULL;
02639                                 }
02640                                 arg = createExprNode(ET_IDENTIFIER, varcopy);
02641                                 if (!arg) {
02642                                         deleteIdentifierNode(name1);
02643                                         deleteIdentifierNode(var);
02644                                         deleteIdentifierNode(varcopy);

02648                                 if (!args) {
02649                                         deleteIdentifierNode(name1);
02650                                         deleteIdentifierNode(var);
02651                                         deleteExprNode(arg);

02669                                 if (!update) {
02670                                         deleteFuncCallExprNode(func);

02672                                 }
02673                         }

02675                                 error("expected unary function", tokens);
02676                                 deleteIdentifierNode(name1);

02678                         }

02681                         if (acceptToken(&tokens, TT_WILE)) {
02682                                 guard = parseExprNode(&tokens, functab);
02683                                 if (!guard) {

02685                                         deleteExprNode(update);
02686                                         return NULL;
02687                                 }
02688                         }
02689                         else if (acceptToken(&tokens, TT_TIL)) {
02690                                 ExprNode *arg = NULL;
02691                                 ExprNodeList *args = NULL;
02692                                 OpExprNode *op = NULL;
02693 #ifdef DEBUG
02694                                 shiftout();
02695                                 debug("ET_OP (OP_NOT)");
02696 #endif
02697                                 arg = parseExprNode(&tokens, functab);
02698                                 if (!arg) {
02699                                         deleteIdentifierNode(name1);
02700                                         deleteExprNode(update);
02701                                         return NULL;
02702                                 }
02703                                 args = createExprNodeList();
02704                                 if (!args) {
02705                                         deleteIdentifierNode(name1);
02706                                         deleteExprNode(update);
02707                                         deleteExprNode(arg);
02708                                         return NULL;
02709                                 }

02711                                         deleteIdentifierNode(name1);
02712                                         deleteExprNode(update);
02713                                         deleteExprNode(arg);
02714                                         deleteExprNodeList(args);
02715                                         return NULL;
02716                                 }

02718                                 shiftin();
02719 #endif

02723                                         deleteExprNode(update);
02724                                         deleteExprNodeList(args);
02725                                         return NULL;
02726                                 }

02728                                 if (!guard) {
02729                                         deleteIdentifierNode(name1);
02730                                         deleteExprNode(update);
02731                                         deleteOpExprNode(op);
02732                                         return NULL;
02733                                 }
02734                         }
02735                 }
02736                 if (!acceptToken(&tokens, TT_NEWLINE)) {
02737                         error("expected end of expression", tokens);
02738                         deleteIdentifierNode(name1);
02739                         if (update) deleteExprNode(update);
02740                         if (guard) deleteExprNode(guard);
02741                         return NULL;
02742                 }
02743                 body = parseBlockNode(&tokens, functab);
02744                 if (!body) {
02745                         deleteIdentifierNode(name1);
02746                         if (update) deleteExprNode(update);
02747                         if (guard) deleteExprNode(guard);
02748                         return NULL;
02749                 }
02750                 if (!acceptToken(&tokens, TT_IMOUTTAYR)) {
02751                         error("expected IM OUTTA YR", tokens);
02752                         deleteIdentifierNode(name1);
02753                         if (update) deleteExprNode(update);
02754                         if (guard) deleteExprNode(guard);
02755                         deleteBlockNode(body);
02756                         return NULL;
02757                 }
02758                 name2 = parseIdentifierNode(&tokens);
02759                 if (!name2) {
02760                         deleteIdentifierNode(name1);
02761                         if (update) deleteExprNode(update);
02762                         if (guard) deleteExprNode(guard);
02763                         deleteBlockNode(body);
02764                         return NULL;
02765                 }

02767                         error("expected matching loop name", tokens);
02768                         deleteIdentifierNode(name1);
02769                         if (update) deleteExprNode(update);
02770                         if (guard) deleteExprNode(guard);
02771                         deleteBlockNode(body);
02772                         deleteIdentifierNode(name2);
02773                         return NULL;
02774                 }
02775                 deleteIdentifierNode(name2);
02776                 if (!acceptToken(&tokens, TT_NEWLINE)) {
02777                         error("expected end of expression", tokens);
02778                         deleteIdentifierNode(name1);
02779                         if (update) deleteExprNode(update);
02780                         if (guard) deleteExprNode(guard);
02781                         deleteBlockNode(body);
02782                         return NULL;

02786                         deleteIdentifierNode(name1);
02787                         if (update) deleteExprNode(update);
02788                         if (guard) deleteExprNode(guard);
02789                         deleteBlockNode(body);
02790                         return NULL;
02791                 }

02793                 if (!ret) {
02794                         deleteLoopStmtNode(stmt);
02795                         return NULL;
02796                 }
02797         }
02798         /* Function definition */
02799         else if (acceptToken(&tokens, TT_HOWDUZ)) {
02800                 IdentifierNode *scope = NULL;
02801                 IdentifierNode *name = NULL;
02802                 BlockNode *body;
02803                 FuncDefStmtNode *proto = NULL;
02804 #ifdef DEBUG
02805                 debug("ST_FUNCDEF");
02806 #endif
02807                 /* Skip over the scope as it has already been parsed */
02808                 scope = parseIdentifierNode(&tokens);
02809                 if (!scope) return NULL;
02810                 deleteIdentifierNode(scope);
02811                 name = parseIdentifierNode(&tokens);
02812                 if (!name) return NULL;
02813                 /* Lookup function prototype to update it after parsing body */
02814                 proto = lookupFuncDefStmtNode(functab, name->image);
02815                 deleteIdentifierNode(name);
02816                 if (!proto) {
02817                         error("function prototype not found", tokens);
02818                         return NULL;
02819                 }
02820                 /* Skip over the arguments as they have already been parsed */
02821                 if (acceptToken(&tokens, TT_YR)) {
02822                         IdentifierNode *arg = parseIdentifierNode(&tokens);
02823                         if (!arg) return NULL;
02824                         deleteIdentifierNode(arg);
02825                         while (acceptToken(&tokens, TT_ANYR)) {
02826                                 arg = parseIdentifierNode(&tokens);
02827                                 if (!arg) return NULL;
02828                                 deleteIdentifierNode(arg);
02829                         }
02830                 }
02831                 if (!acceptToken(&tokens, TT_NEWLINE)) {
02832                         error("expected end of statement", tokens);
02833                         return NULL;
02834                 }
02835                 body = parseBlockNode(&tokens, functab);
02836                 if (!body) return NULL;
02837                 if (!acceptToken(&tokens, TT_IFUSAYSO)) {
02838                         error("expected IF YOU SAY SO", tokens);
02839                         deleteBlockNode(body);
02840                         return NULL;
02841                 }
02842                 if (!acceptToken(&tokens, TT_NEWLINE)) {
02843                         error("expected end of statement", tokens);
02844                         deleteBlockNode(body);
02845                         return NULL;
02846                 }
02847                 /* Update the function body */
02848                 proto->body = body;
02849                 ret = createStmtNode(ST_FUNCDEF, proto);
02850                 if (!ret) {
02851                         deleteStmtNode(ret);
02852                 }
02853         }
02854         /* Expression Evaluation */
02855         else if ((expr = parseExprNode(&tokens, functab))) {
02856 #ifdef DEBUG
02857                 debug("ST_EXPR");
02858 #endif
02859                 if (!acceptToken(&tokens, TT_NEWLINE)) {
02860                         deleteExprNode(expr);
02861                         error("expected end of expression", tokens);
02862                         return NULL;
02863                 }
02864                 ret = createStmtNode(ST_EXPR, expr);
02865         }
02866         else error("expected statement", tokens);
02867 #ifdef DEBUG
02868         shiftin();
02869 #endif
02870         *tokenp = tokens;
02871         return ret;
02872 }

02892 BlockNode *parseBlockNode(Token ***tokenp,        
02893                           FunctionTable *functab) 
02894 {
02895         Token **tokens = *tokenp;
02896         StmtNodeList *stmts = NULL;
02897         BlockNode *block = NULL;
02898 #ifdef DEBUG
02899         shiftout();
02900         debug(">ET_BLOCK");
02901 #endif
02902         stmts = createStmtNodeList();
02903         if (!stmts) return NULL;
02904         while (!peekToken(&tokens, TT_EOF)
02905                         && !peekToken(&tokens, TT_KTHXBYE) && !peekToken(&tokens, TT_OIC)
02906                         && !peekToken(&tokens, TT_YARLY) && !peekToken(&tokens, TT_NOWAI)
02907                         && !peekToken(&tokens, TT_MEBBE) && !peekToken(&tokens, TT_OMG)
02908                         && !peekToken(&tokens, TT_OMGWTF) && !peekToken(&tokens, TT_IMOUTTAYR)
02909                         && !peekToken(&tokens, TT_IFUSAYSO)) {
02910                 StmtNode *s = parseStmtNode(&tokens, functab);
02911                 if (!s) {
02912                         deleteStmtNodeList(stmts);
02913                         return NULL;
02914                 }
02915                 addStmtNode(stmts, s);
02916         }
02917 #ifdef DEBUG
02918         debug("<ET_BLOCK");
02919         shiftin();
02920 #endif
02921         *tokenp = tokens;
02922         block = createBlockNode(stmts);
02923         if (!block) {
02924                 deleteStmtNodeList(stmts);
02925                 return NULL;
02926         }
02927         return block;
